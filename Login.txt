User class

package com.aurionpro.model;

import com.aurionpro.exception.EmailNotValidException;
import com.aurionpro.exception.PasswordNotValidException;

public class User {
	
private String email;
private String password;

public User(String email, String password) {
	super();
	if(!isValidEmail(email))
		throw new EmailNotValidException("The email must be alphanumeric.It must contain atleast one special character.");	
	this.email = email;
	if( !isValidPassword(password))
		throw new PasswordNotValidException("Password must be at least 8 characters long and include at least one uppercase, one lowercase, one digit, and one special character.");
	this.password = password;
}

private boolean isValidEmail(String email) {
    
    return email != null && email.matches(".+@.+\\..+");
}

private boolean isValidPassword(String password) {
   
    return password != null && password.length() >= 8 &&
           password.matches("^(?=.*[0-9])(?=.*[a-z])(?=.*[A-Z])(?=.*[!@#&()â€“[{}]:;',?/*~$^+=<>]).{8,}$");
}


@Override
public String toString() {
	return "User [email=" + email + ", password=" + password + "]";
}

}
